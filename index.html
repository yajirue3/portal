document.addEventListener("DOMContentLoaded", async () => { const container = document.getElementById("sites"); try { const res = await fetch("sites.json"); const sites = await res.json(); container.innerHTML = ""; sites.forEach(site => { const div = document.createElement("div"); div.className = "site"; div.innerHTML = ` ${site.name} `; container.appendChild(div); }); } catch (e) { console.error(e); container.innerHTML = "読み込み失敗"; } });
